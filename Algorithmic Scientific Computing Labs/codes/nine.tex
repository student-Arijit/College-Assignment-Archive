#include <stdio.h>
#define N 3

int main() {
    float a[N][N+1], x[N];
    int i,j,k;
    float ratio;

    printf("Enter augmented matrix (%dx%d):\n", N, N+1);
    for(i=0;i<N;i++)
        for(j=0;j<=N;j++)
            scanf("%f",&a[i][j]);

    for(i=0;i<N-1;i++){
        for(j=i+1;j<N;j++){
            ratio=a[j][i]/a[i][i];
            for(k=0;k<=N;k++) a[j][k]-=ratio*a[i][k];
        }
    }

    x[N-1]=a[N-1][N]/a[N-1][N-1];
    for(i=N-2;i>=0;i--){
        x[i]=a[i][N];
        for(j=i+1;j<N;j++) x[i]-=a[i][j]*x[j];
        x[i]/=a[i][i];
    }

    printf("Solution:\n");
    for(i=0;i<N;i++) printf("x[%d]=%f\n",i+1,x[i]);
    return 0;
}
