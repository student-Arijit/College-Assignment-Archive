#include <stdio.h>
#include <math.h>

#define EPS 1e-6
#define MAX_ITER 100

double f(double x) { return x*x - 4*x - 10; }
double df(double x) { return 2*x - 4; }

int main() {
    double x0 = 5.0, x1;
    int iter = 0;

    do {
        if (fabs(df(x0)) < EPS) {
            printf("Derivative too small. Choose another initial 
            guess.\n");
            return 1;
        }
        x1 = x0 - f(x0)/df(x0);
        if (fabs(x1 - x0) < EPS) break;
        x0 = x1;
        iter++;
    } while(iter < MAX_ITER);

    printf("Root = %lf\n", x1);
    return 0;
}


\end{verbatim}
