a) Trapezoidal Rule:
#include <stdio.h>
#include <math.h>

double f(double x) { return 1/(1+x*x); }

int main() {
    int n, i;
    double a, b, h, sum=0;
    printf("Enter a, b, n: ");
    scanf("%lf %lf %d", &a, &b, &n);
    h = (b-a)/n;
    sum = f(a)+f(b);
    for(i=1; i<n; i++) sum += 2*f(a+i*h);
    sum = sum*h/2;
    printf("Trapezoidal = %lf\n", sum);
    return 0;
}

b) Simpson’s 1/3 Rule:
#include <stdio.h>
#include <math.h>

double f(double x) { return 1/(1+x*x); }

int main() {
    int n, i;
    double a, b, h, sum=0;
    printf("Enter a, b, n (even): ");
    scanf("%lf %lf %d", &a, &b, &n);
    h = (b-a)/n;
    sum = f(a)+f(b);
    for(i=1; i<n; i++)
        sum += (i%2==0 ? 2 : 4)*f(a+i*h);
    sum = sum*h/3;
    printf("Simpson 1/3 = %lf\n", sum);
    return 0;
}
