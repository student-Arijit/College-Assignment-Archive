#include <stdio.h>
#include <math.h>

#define N 3
#define EPS 1e-6

int main() {
    float a[N][N+1] = {{4, -1, 0, 3},
                       {-1, 4, -1, 8},
                       {0, -1, 4, -6}};
    float x[N] = {0}, old_x[N];
    int i, j, iter = 0;
    float sum, error;

    do {
        iter++;
        for(i = 0; i < N; i++) old_x[i] = x[i];

        for(i = 0; i < N; i++) {
            sum = a[i][N];
            for(j = 0; j < N; j++) {
                if(j != i) sum -= a[i][j] * x[j];
            }
            x[i] = sum / a[i][i];
        }

        error = 0;
        for(i = 0; i < N; i++)
            error += fabs(x[i] - old_x[i]);
    } while(error > EPS);

    printf("Solution after %d iterations:\n", iter);
    for(i = 0; i < N; i++) printf("x[%d] = %f\n", i+1, x[i]);
    return 0;
}
