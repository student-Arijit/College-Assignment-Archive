    public class Square {
    protected double side;

    public Square(double side) {
        this.side = side;
    }

    public double getVolume() {
        double volume = side * side * side;  // 
        Treating it as cube volume
        System.out.println("Volume of Square (as 
        Cube) = " + volume);
        return volume;
    }
}
public class Cylinder extends Square {
    private double height;

    public Cylinder(double radius, double height) {
        super(radius);  // 'side' of Square 
        becomes 'radius' of Cylinder
        this.height = height;
    }
    @Override
    public double getVolume() {
        double volume = Math.PI * side * side * 
        height;  // pi * r² * h
        System.out.println("Volume of Cylinder = 
        " + volume);
        return volume;
    }
}
public class TestShape {
    public static void main(String[] args) {
        // Create Square object
        Square cube = new Square(3);
        cube.getVolume();  // 27.0

        // Create Cylinder object
        Cylinder cyl = new Cylinder(3, 5);
        cyl.getVolume();   // Should print 
        Cylinder volume using radius = 3, height = 5
    }
}
