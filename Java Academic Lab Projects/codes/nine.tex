import java.sql.*;
import java.util.Scanner;
public class StudentJDBC {
    public static void main(String[] args) {
        String url = "jdbc:mysql://localhost:3306/your_database
        "; // change DB name
        String user = "your_username"; // your DB username
        String password = "your_password"; // your DB password
        Scanner sc = new Scanner(System.in);
        try {
            // Load JDBC Driver (optional for modern JDBC)
            Class.forName("com.mysql.cj.jdbc.Driver");
            // Establish connection
            Connection conn = DriverManager.getConnection(url, 
            user, password);
            // Take input from user
            System.out.print("Enter Reg No: ");
            int regno = sc.nextInt();
            sc.nextLine(); // consume newline
            System.out.print("Enter Student Name: ");
            String sname = sc.nextLine();
            System.out.print("Enter City: ");
            String city = sc.nextLine();
            System.out.print("Enter Contact No: ");
            String contactno = sc.nextLine();
            // Insert into table
            String insertSQL = "INSERT INTO Student_info (regno, 
            sname, city, contactno) VALUES (?, ?, ?, ?)";
            PreparedStatement pstmt = conn.prepareStatement(
            insertSQL);
            pstmt.setInt(1, regno);
            pstmt.setString(2, sname);
            pstmt.setString(3, city);
            pstmt.setString(4, contactno);
            pstmt.executeUpdate();
            System.out.println("\nStudent data 
            inserted successfully!\n");
            // Display all records
            Statement stmt = conn.createStatement();
            ResultSet rs = stmt.executeQuery("SELECT * FROM
            Student_info");
            System.out.printf("%-10s %-20s %-15s %-15s\n",
            "RegNo", "Name", "City", "ContactNo");
            System.out.println("-------------------------
            -----------------");
            while (rs.next()) {
                System.out.printf("%-10d %-20s %-15s %-15s\n",
                        rs.getInt("regno"),
                        rs.getString("sname"),
                        rs.getString("city"),
                        rs.getString("contactno"));
            }
            // Close resources
            rs.close();
            stmt.close();
            pstmt.close();
            conn.close();
        } catch (ClassNotFoundException e) {
            System.out.println("JDBC Driver not found.");
        } catch (SQLException e) {
            System.out.println("Database error: " + 
            e.getMessage());
        } finally {
            sc.close();
        }
    }
}
