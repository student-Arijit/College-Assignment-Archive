    public class ExceptionHandlingDemo {
    public static void main(String[] args) {
        int[] numbers = {10, 20, 30};
        int result = 0;
        try {
            // This will cause ArrayIndexOutOfBoundsException
            int value = numbers[5];

            // This will cause ArithmeticException (division 
            by zero)
            result = value / 0;

            System.out.println("Result: " + result);

        } catch (ArrayIndexOutOfBoundsException e) {
            System.out.println("Error: Tried to 
            access an invalid index in the array.");
        } catch (ArithmeticException e) {
            System.out.println("Error: Division 
            by zero is not allowed.");
        }

        System.out.println("Program continues 
        normally after exception handling.");
    }
}
\end{verbatim}
